using System.ComponentModel.DataAnnotations;

namespace systemd_service_monitor.Configuration
{
    public class SystemdServiceMonitorOptions
    {
        [Required]
        public SystemdOptions Systemd { get; set; } = new SystemdOptions();

        [Required]
        public DatabaseOptions Database { get; set; } = new DatabaseOptions();

        [Required]
        public SwaggerOptions Swagger { get; set; } = new SwaggerOptions();

        [Required]
        public AlertOptions Alerts { get; set; } = new AlertOptions();

        public class SystemdOptions
        {
            [Required]
            public bool EnableMonitoring { get; set; } = true;

            [Range(1000, 60000)]
            public int MetricCollectionIntervalMs { get; set; } = 5000;

            [Range(1, 365)]
            public int LogRetentionDays { get; set; } = 30;

            [Range(1, 10000)]
            public int MaxLogEntriesPerRequest { get; set; } = 1000;

            public bool EnableRemoteOperations { get; set; } = true;

            [Range(1000, 60000)]
            public int OperationTimeoutMs { get; set; } = 30000;

            [Range(1, 10)]
            public int ConnectionRetryCount { get; set; } = 3;

            [Range(100, 5000)]
            public int ConnectionRetryDelayMs { get; set; } = 1000;

            public bool EnableHealthChecks { get; set; } = true;
        }

        public class DatabaseOptions
        {
            [Required]
            public string ConnectionString { get; set; } = string.Empty;

            [Required]
            [EnumDataType(typeof(DatabaseProvider))]
            public string Provider { get; set; } = "InMemory";

            public bool EnableLogging { get; set; } = false;

            [Range(1, 300)]
            public int CommandTimeoutSeconds { get; set; } = 30;

            [Range(1, 100)]
            public int MaxConnectionPoolSize { get; set; } = 20;

            [Range(1, 60)]
            public int QueryCacheExpirationMinutes { get; set; } = 5;
        }

        public enum DatabaseProvider
        {
            InMemory,
            SqlServer,
            PostgreSql
        }

        public class SwaggerOptions
        {
            public bool Enabled { get; set; } = true;

            [Required]
            public string Title { get; set; } = "systemd Service Monitor API";

            [Required]
            public string Version { get; set; } = "1.0.0";

            [Required]
            public string Description { get; set; } = "REST API for monitoring and managing systemd services via D-Bus";
        }

        public class AlertOptions
        {
            public bool Enabled { get; set; } = true;

            public bool AutoResolveOnConditionCleared { get; set; } = true;

            [Range(1000, 1000000)]
            public int MaxIncidentHistorySize { get; set; } = 10000;

            [Range(10, 3600)]
            public int EscalationCheckIntervalSeconds { get; set; } = 60;

            [Range(5, 60)]
            public int ServiceEvaluationIntervalSeconds { get; set; } = 30;

            [Range(0, 300)]
            public int StartupDelaySeconds { get; set; } = 10;

            public WebhookOptions Webhook { get; set; } = new WebhookOptions();

            public EscalationDefaults EscalationDefaults { get; set; } = new EscalationDefaults();
        }

        public class WebhookOptions
        {
            [Range(1, 30)]
            public int TimeoutSeconds { get; set; } = 10;

            [Range(1, 5)]
            public int MaxRetries { get; set; } = 3;

            [Range(100, 5000)]
            public int RetryDelayMs { get; set; } = 1000;

            public Dictionary<string, string> DefaultHeaders { get; set; } = new Dictionary<string, string>();
        }

        public class EscalationDefaults
        {
            [Range(1, 60)]
            public int InitialEscalationDelayMinutes { get; set; } = 5;

            [Range(1, 1440)]
            public int SubsequentEscalationDelayMinutes { get; set; } = 15;

            [Range(1, 10)]
            public int MaxEscalationLevels { get; set; } = 4;

            [Range(1, 1440)]
            public int DefaultCooldownMinutes { get; set; } = 15;
        }
    }
}
